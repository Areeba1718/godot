96 // Read 16, 24 or 32 bits stored in little-endian order.
97 static WEBP_INLINE int GetLE16(const uint8_t* const data) {
98  return (int)(data[0] << 0) | (data[1] << 8);
99 }
100 
101 static WEBP_INLINE int GetLE24(const uint8_t* const data) {
102  return GetLE16(data) | (data[2] << 16);
103 }
104 
105 static WEBP_INLINE uint32_t GetLE32(const uint8_t* const data) {
106  return (uint32_t)GetLE16(data) | (GetLE16(data + 2) << 16);
107 }

109 // Store 16, 24 or 32 bits in little-endian order.
110 static WEBP_INLINE void PutLE16(uint8_t* const data, int val) {
111  assert(val < (1 << 16));
112  data[0] = (val >> 0);
113  data[1] = (val >> 8);
114 }
115 
116 static WEBP_INLINE void PutLE24(uint8_t* const data, int val) {
117  assert(val < (1 << 24));
118  PutLE16(data, val & 0xffff);
119  data[2] = (val >> 16);
120 }
121 
122 static WEBP_INLINE void PutLE32(uint8_t* const data, uint32_t val) {
123  PutLE16(data, (int)(val & 0xffff));
124  PutLE16(data + 2, (int)(val >> 16));
125 }
