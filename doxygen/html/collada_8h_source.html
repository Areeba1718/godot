333  bool operator<(const Vertex& p_vert) const {
334 
335  if (uid==p_vert.uid) {
336  if (vertex==p_vert.vertex) {
337  if(normal==p_vert.normal) {
338  if(uv==p_vert.uv) {
339  if(uv2==p_vert.uv2) {
340 
341  if (!weights.empty() || !p_vert.weights.empty()) {

343  if (weights.size()==p_vert.weights.size()) {

346  if (weights[i].bone_idx!=p_vert.weights[i].bone_idx)
347  return weights[i].bone_idx<p_vert.weights[i].bone_idx;
348 
349  if (weights[i].weight!=p_vert.weights[i].weight)
350  return weights[i].weight<p_vert.weights[i].weight;
351  }
352  } else {
353  return weights.size() < p_vert.weights.size();
354  }
