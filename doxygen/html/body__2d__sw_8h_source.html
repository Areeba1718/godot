283 void Body2DSW::add_contact(const Vector2& p_local_pos,const Vector2& p_local_normal, float p_depth, int p_local_shape, const Vector2& p_collider_pos, int p_collider_shape, ObjectID p_collider_instance_id, const RID& p_collider,const Vector2& p_collider_velocity_at_pos) {
284 
285  int c_max=contacts.size();
286 
287  if (c_max==0)
288  return;
289 
290  Contact *c = &contacts[0];
291 
292 
293  int idx=-1;
294 
295  if (contact_count<c_max) {
296  idx=contact_count++;
297  } else {
298 
299  float least_depth=1e20;
300  int least_deep=-1;
301  for(int i=0;i<c_max;i++) {
302 
303  if (i==0 || c[i].depth<least_depth) {
304  least_deep=i;
305  least_depth=c[i].depth;
306  }
307  }
308 
309  if (least_deep>=0 && least_depth<p_depth) {
310 
311  idx=least_deep;
312  }
313  if (idx==-1)
314  return; //none least deepe than this
315  }
316 
317  c[idx].local_pos=p_local_pos;
318  c[idx].local_normal=p_local_normal;
319  c[idx].depth=p_depth;
320  c[idx].local_shape=p_local_shape;
321  c[idx].collider_pos=p_collider_pos;
322  c[idx].collider_shape=p_collider_shape;
323  c[idx].collider_instance_id=p_collider_instance_id;
324  c[idx].collider=p_collider;
325  c[idx].collider_velocity_at_pos=p_collider_velocity_at_pos;
