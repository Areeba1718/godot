431 
432  if (Thread::get_main_ID()!=Thread::get_caller_ID())
433  return; //no support for other threads than main for now

435  if (ScriptDebugger::get_singleton()->get_lines_left()>0 && ScriptDebugger::get_singleton()->get_depth()>=0)
436  ScriptDebugger::get_singleton()->set_depth( ScriptDebugger::get_singleton()->get_depth() +1 );
437 
438  if (_debug_call_stack_pos >= _debug_max_call_stack) {

448  _call_stack[_debug_call_stack_pos].ip=p_ip;
