52 /* The list of feature flags for ARM CPUs that can be recognized by the
53  * library. Value details are:
54  *
55  * VFPv2:
56  * CPU supports the VFPv2 instruction set. Many, but not all, ARMv6 CPUs
57  * support these instructions. VFPv2 is a subset of VFPv3 so this will
58  * be set whenever VFPv3 is set too.
59  *
60  * ARMv7:
61  * CPU supports the ARMv7-A basic instruction set.
62  * This feature is mandated by the 'armeabi-v7a' ABI.
63  *
64  * VFPv3:
65  * CPU supports the VFPv3-D16 instruction set, providing hardware FPU
66  * support for single and double precision floating point registers.
67  * Note that only 16 FPU registers are available by default, unless
68  * the D32 bit is set too. This feature is also mandated by the
69  * 'armeabi-v7a' ABI.
70  *
71  * VFP_D32:
72  * CPU VFP optional extension that provides 32 FPU registers,
73  * instead of 16. Note that ARM mandates this feature is the 'NEON'
74  * feature is implemented by the CPU.
75  *
76  * NEON:
77  * CPU FPU supports "ARM Advanced SIMD" instructions, also known as
78  * NEON. Note that this mandates the VFP_D32 feature as well, per the
79  * ARM Architecture specification.
80  *
81  * VFP_FP16:
82  * Half-width floating precision VFP extension. If set, the CPU
83  * supports instructions to perform floating-point operations on
84  * 16-bit registers. This is part of the VFPv4 specification, but
85  * not mandated by any Android ABI.
86  *
87  * VFP_FMA:
88  * Fused multiply-accumulate VFP instructions extension. Also part of
89  * the VFPv4 specification, but not mandated by any Android ABI.
90  *
91  * NEON_FMA:
92  * Fused multiply-accumulate NEON instructions extension. Optional
93  * extension from the VFPv4 specification, but not mandated by any
94  * Android ABI.
95  *
96  * IDIV_ARM:
97  * Integer division available in ARM mode. Only available
98  * on recent CPUs (e.g. Cortex-A15).
99  *
100  * IDIV_THUMB2:
101  * Integer division available in Thumb-2 mode. Only available
102  * on recent CPUs (e.g. Cortex-A15).
103  *
104  * iWMMXt:
105  * Optional extension that adds MMX registers and operations to an
106  * ARM CPU. This is only available on a few XScale-based CPU designs
107  * sold by Marvell. Pretty rare in practice.
108  *
109  * If you want to tell the compiler to generate code that targets one of
110  * the feature set above, you should probably use one of the following
111  * flags (for more details, see technical note at the end of this file):
112  *
113  * -mfpu=vfp
114  * -mfpu=vfpv2
115  * These are equivalent and tell GCC to use VFPv2 instructions for
116  * floating-point operations. Use this if you want your code to
117  * run on *some* ARMv6 devices, and any ARMv7-A device supported
118  * by Android.
119  *
120  * Generated code requires VFPv2 feature.
121  *
122  * -mfpu=vfpv3-d16
123  * Tell GCC to use VFPv3 instructions (using only 16 FPU registers).
124  * This should be generic code that runs on any CPU that supports the
125  * 'armeabi-v7a' Android ABI. Note that no ARMv6 CPU supports this.
126  *
127  * Generated code requires VFPv3 feature.
128  *
129  * -mfpu=vfpv3
130  * Tell GCC to use VFPv3 instructions with 32 FPU registers.
131  * Generated code requires VFPv3|VFP_D32 features.
132  *
133  * -mfpu=neon
134  * Tell GCC to use VFPv3 instructions with 32 FPU registers, and
135  * also support NEON intrinsics (see <arm_neon.h>).
136  * Generated code requires VFPv3|VFP_D32|NEON features.
137  *
138  * -mfpu=vfpv4-d16
139  * Generated code requires VFPv3|VFP_FP16|VFP_FMA features.
140  *
141  * -mfpu=vfpv4
142  * Generated code requires VFPv3|VFP_FP16|VFP_FMA|VFP_D32 features.
143  *
144  * -mfpu=neon-vfpv4
145  * Generated code requires VFPv3|VFP_FP16|VFP_FMA|VFP_D32|NEON|NEON_FMA
146  * features.
147  *
148  * -mcpu=cortex-a7
149  * -mcpu=cortex-a15
150  * Generated code requires VFPv3|VFP_FP16|VFP_FMA|VFP_D32|
151  * NEON|NEON_FMA|IDIV_ARM|IDIV_THUMB2
152  * This flag implies -mfpu=neon-vfpv4.
153  *
154  * -mcpu=iwmmxt
155  * Allows the use of iWMMXt instrinsics with GCC.
156  */
157 enum {
158  ANDROID_CPU_ARM_FEATURE_ARMv7 = (1 << 0),

160  ANDROID_CPU_ARM_FEATURE_NEON = (1 << 2),

184 /* The following is used to force the CPU count and features
