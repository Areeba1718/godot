42 // GetURLHandler is used to download data from |url|. When download is
43 // finished or when an error occurs, it posts a message back to the browser
44 // with the results encoded in the message as a string and self-destroys.
45 //
46 // EXAMPLE USAGE:
47 // GetURLHandler* handler* = GetURLHandler::Create(instance,url);
48 // handler->Start();
49 //
50 class GetURLHandler {
51

55 
56  STATUS_NONE,
57  STATUS_IN_PROGRESS,
58  STATUS_COMPLETED,
59  STATUS_ERROR,
60  };
61 
62 private:
63 
64  Status status;
65 
66  // Callback fo the pp::URLLoader::Open().
67  // Called by pp::URLLoader when response headers are received or when an
68  // error occurs (in response to the call of pp::URLLoader::Open()).
69  // Look at <ppapi/c/ppb_url_loader.h> and
70  // <ppapi/cpp/url_loader.h> for more information about pp::URLLoader.
71  void OnOpen(int32_t result);
72 
73  // Callback fo the pp::URLLoader::ReadResponseBody().
74  // |result| contains the number of bytes read or an error code.
75  // Appends data from this->buffer_ to this->url_response_body_.
76  void OnRead(int32_t result);
77 
78  // Reads the response body (asynchronously) into this->buffer_.

89  pp::URLLoader url_loader_; // URLLoader provides an API to download URLs.

93  bool complete;

95  GetURLHandler(const GetURLHandler&);
